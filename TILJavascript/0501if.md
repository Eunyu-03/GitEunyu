## 만들어 본 코드 - if

```Javascript
let A=10;
let B=5;

console.log("A의 값: ", A, ',', "B의 값: ", B);

if(A>B){
    console.log('A가 더 크다.');
}

else if(B<A){
    console.log('B가 더 크다.');
}

else{
    console.log('두 수가 같다.')
}
```
